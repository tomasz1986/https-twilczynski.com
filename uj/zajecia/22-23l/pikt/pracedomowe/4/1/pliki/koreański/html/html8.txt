            function setCookie(cName, cValue, cDay) {
                var expire = new Date();
                expire.setDate(expire.getDate() + cDay);
                cookies = cName + '=' + (cValue) + '; path=/ ; domain=' + '.mt.co.kr'; // 한글 깨짐을 막기위해 escape(cValue)를 합니다.
                if (typeof cDay != 'undefined') cookies += ';expires=' + expire.toGMTString() + ';';
                document.cookie = cookies;
            }

            // 쿠키 가져오기
            function getCookie(cName) {
                cName = cName + '=';
                var cookieData = document.cookie;
                var start = cookieData.indexOf(cName);
                var cValue = '';

                if (start != -1) {
                    start += cName.length;
                    var end = cookieData.indexOf(';', start);
                    if (end == -1) end = cookieData.length;
                    cValue = cookieData.substring(start, end);
                }
                return unescape(cValue);
            }

                    $("#negativeCnt").html(serverData[1] + '%');

                    if (param == "POSITIVE") {
                        if ($('.pos_btn').hasClass('on')) {
                            $('.pos_btn').removeClass('on');
                        } else {
                            $('.pos_btn').addClass('on');
                            $('.neg_btn').removeClass('on');

                            alert("공감하셨습니다.");
                        }
                        var agNoStr = getCookie("agflag");
                        var agNoArr = null;

                        if (typeof (agNoStr) != "undefined" && agNoStr) {
                            agNoArr = agNoStr.split('|');
                            var noIndex = agNoArr.indexOf(gisa_id);

                            if (undoCnt == 0 && noIndex == -1) {
                                if (agNoArr.length >= 100) {
                                    agNoArr.splice(0, 1);
                                }
                                agNoArr.push(gisa_id);
                            } else if (undoCnt == 1 && noIndex != -1) {
                                agNoArr.splice(noIndex, 1);
                            }
                        } else {
                            if (undoCnt == 0) {
                                agNoArr = new Array();
                                agNoArr.push(gisa_id);
                            }
                        }

                        if (agNoArr.length <= 0) {
                            setCookie("agflag", "", "-1");
                        } else {
                            agNoStr = agNoArr.join('|');
                            setCookie("agflag", agNoStr, "10000");
                        }
                    } else { // NEGATIVE
                        if ($('.neg_btn').hasClass('on')) {
                            $('.neg_btn').removeClass('on');
                        } else {
                            $('.neg_btn').addClass('on');
                            $('.pos_btn').removeClass('on');

                            alert("비공감하셨습니다.");
                        }
                        var disagNoStr = getCookie("disagflag");
                        var disagNoArr = null;

                        if (typeof (disagNoStr) != "undefined" && disagNoStr) {
                            disagNoArr = disagNoStr.split('|');
                            var noIndex = disagNoArr.indexOf(gisa_id);

                            if (undoCnt == 0 && noIndex == -1) {
                                if (disagNoArr.length >= 100) {
                                    disagNoArr.splice(0, 1);
                                }
                                disagNoArr.push(gisa_id);
                            } else if (undoCnt == 1 && noIndex != -1) {
                                disagNoArr.splice(noIndex, 1);
                            }
                        } else {
                            if (undoCnt == 0) {
                                disagNoArr = new Array();
                                disagNoArr.push(gisa_id);
                            }
                        }

                        if (disagNoArr.length <= 0) {
                            setCookie("disagflag", "", "-1");
                        } else {
                            disagNoStr = disagNoArr.join('|');
                            setCookie("disagflag", disagNoStr, "10000");
                        }
                    }
                }
            }
        });
